int bin2(int n, int k) {

	int i, j;


	int **b;
	b = (int**)malloc(sizeof(int*) * n);
	for (int i = 0; i < n; i++) {
		b[i] = (int*)malloc(sizeof(int) * k);
	}

	//int b[100][100];



	for (i = 0; i <= n; i++) {

		for (j = 0; j <= MIN(i, k); j++) {
			if (j == 0 || j == i) b[i][j] = 1;
			else b[i][j] = b[i - 1][j - 1] + b[i - 1][j];

		}

	}

	return b[n][k];

}
