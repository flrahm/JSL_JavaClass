void floyd(int n , const int W[][6] , int D[][6] ) {
	

	for (int i = 1; i <= 5; i++)
		for (int j = 1; j <= 5; j++)
			D[i][j] = W[i][j];



	for (int k = 1; k <= n; k++)
		for (int i = 1; i <= n; i++)
			for (int j = 1; j <= n; j++)
				D[i][j] = MIN(D[i][j], D[i][k] + D[k][j]);




}


void floyd2(int n, const int W[][6], int D[][6], int P[][6]) {
	
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= n; j++)
			P[i][j] = 0;

	for (int i = 1; i <= 5; i++)
		for (int j = 1; j <= 5; j++)
			D[i][j] = W[i][j];


	for (int k = 1; k <= n; k++)
		for (int i = 1; i <= n; i++)
			for (int j = 1; j <= n; j++)
				if ((D[i][k] + D[k][j]) < D[i][j]) {
					P[i][j] = k;
					D[i][j] = D[i][k] + D[k][j];
				}

}



void path(int q, int r,int P[][6]) {


	

	if (P[q][r] != 0) {
		
		path(q, P[q][r],P);
		cout << " -> " << P[q][r];
		path(P[q][r], r,P);
	}

	
}
